        EAX
        EBX 
        ECX
        EDX
    }
    16 bits registers: {
        AX
        BX
        CX
        DX
    }
    8 bits registers: {
        AL
        BL
        CL
        DL
    }
}

    64 bits registers: {
        RSI
        RDI
    }

    32 bits registers: {
        ESI
        EDI
    }

    16 bits registers: {
        SI
        DI
    }
}

    Sign Flag (SF)
