Create file: {
    rax - CREAT syscall number
    rdi - file name
    rsi - file mode, 0777q is good

    rax - receive file decriptor
}

Open file: {
    rax - OPEN syscall number
    rdi - file name
    rsi - open mode (0 = read)
    rdx - file mode

    rax - receive file decriptor
}

Write file: {
    rax - WRITE syscall number
    rdi - file decriptor
    rsi - text to write
    rdx - size of write
}

Read file: {
    rax - READ syscall number
    rdi - file decriptor
    rsi - place of read
}

Close file: {
    rax - CLOSE syscall number
    rdi - file decriptor
}
